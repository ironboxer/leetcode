### [Length of Last Word](https://leetcode.com/problems/length-of-last-word/)

### String


```Go
func lengthOfLastWord(s string) int {
	if len(s) == 0 {
        return 0
    }
    left, right := 0, len(s) - 1
    for ; right >= 0 && s[right] == ' '; right-- {}
    for ; left <= right && s[left] == ' '; left++ {}
    space := left - 1
    for i := left; i <= right; i++ {
        e := s[i]
        if e == ' ' {
            space = i
        }
    }
    return right - space
}
```
