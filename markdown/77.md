### [Combinations](https://leetcode.com/problems/combinations/)


### Combination


```Python
class Solution:
    def combine(self, n: int, k: int) -> List[List[int]]:
        res = []
        def f(start, n, k, tmp):
            if len(tmp) == k:
                nonlocal res
                res.append(tmp[:])
                return
            for i in range(start, n + 1):
                tmp.append(i)
                f(i+1, n, k, tmp)
                tmp.pop()

        f(1, n, k, [])
        return res
```

    res := make([][]int, 0)
    var f func(start int, tmp []int)
    f = func(start int, tmp []int) {
        if len(tmp) == k {
			cp := make([]int, len(tmp))

            res = append(res, cp)
			return
		}
        for i := start; i <= n; i++ {
            tmp = append(tmp, i)
            f(i+1, tmp)
            tmp = tmp[:len(tmp)-1]
        }
    }
	tmp := make([]int, 0)
    f(1, tmp)
    return res
}
